corrs = rep(NULL, n_iter)

Package approach

foo = function(data, indexes) {
    return(cor(data[indexes,1],data[indexes,2], method = "pearson"))

results = boot(dt, foo, R = n_iter, stype = 'i' )
lmd = boot.ci(boot.out = results, type = "norm", conf=0.95)

Compare with the Fisher transform method.

Fisher z-transform

The sampling distribution of Pearson r is not normally distributed. Fisher developed a transformation now called “Fisher z-transformation” that converts Pearson r to the normally distributed variable z. The formula for the transformation is: \[z_r = tanh^{-1}(r) = \frac{1}{2}log\left ( \frac{1+r}{1-r}\right )\]

r = cor(X,Y, method = "pearson")
z = FisherZ(r)
se = (1.96)*(1/sqrt(N-3))
fzt = round(FisherZInv(c(z, z - se, z + se)), 3)

names(fzt) = c('Estimate',   'CI lower',   'CI upper')
fzt
## Estimate CI lower CI upper 
##    0.049   -0.064    0.162
